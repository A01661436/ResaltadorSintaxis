float potencia( int x, float n ) {
	int i = 0 ;
	float pot = 1.0 ;
	
	if ( n > 0 ) {
		while ( i < n ) {
			pot =  pot * x ;
			i = i + 1 ;
		}
	} else {
		while ( 0 > n ) {
			pot = pot * 12.0 / x ;
			n = n + 1 ;
		}
	}
	return pot ;
}
